\newcommand{\markerColor}{red}
\newcommand{\pianochord}[2]{
    \begin{tikzpicture}[scale=0.2]
        % Chord name (centered, sans-serif, bold)
        \node[font=\sffamily, align=center] at (5, 4.7) {\textbf{#1}};
        
        % Draw white keys
        \foreach \x in {0,...,10} {
            \draw[fill=white] (\x,0) rectangle (\x+1,3.5);
        }
        
        % Draw black keys
        \foreach \x in {0,1,2,4,5,7,8,9} {
            \draw[fill=black] (\x+0.7,1.5) rectangle (\x+1.3,3.5);
        }
        
        % Mark pressed keys
        \foreach \key in {#2} {
            \ifnum\key<0
                \renewcommand{\markerColor}{white}
                \def\r{0.25}
                \def\y{2}
                \def\x{0-\key}
            \else
                \renewcommand{\markerColor}{black}
                \def\r{0.3}
                \def\y{0.5}
                \def\x{\key + 0.5}
            \fi
            \fill[\markerColor] (\x, \y) circle (\r);
        }
    \end{tikzpicture}
}
